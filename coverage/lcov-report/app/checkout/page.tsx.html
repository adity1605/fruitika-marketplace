    country: "",
    phone: "",
    cardNumber: "",
    expiryDate: "",

    script.src = 'https://checkout.razorpay.com/v1/checkout.js'
    script.async = true
    document.body.appendChild(script)
    return () => {
      document.body.removeChild(script)
    }

    const options = {
